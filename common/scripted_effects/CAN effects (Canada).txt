CAN_transfer_iedc_leadership = {
	if = {
		limit = {
			OR = {
				is_in_economic_sphere = no
				AND = {
					has_event_target = iedc_leader
					event_target:iedc_leader = { is_leader_of_economic_sphere_of_PREV = yes }
				}
			}
		}
		CAN_add_to_iedc = yes
		if = {
			limit = { has_event_target = iedc_leader }
			event_target:iedc_leader = { remove_ideas = CAN_iedc_main }
		}
		add_ideas = CAN_iedc_main
		set_economic_sphere_leader = yes
	}
}

CAN_find_entente_leader = {
	if = {
		limit = {
			NOT = { tag = GBR }
			GBR = {
				is_in_faction_with = ROOT
				is_subject = no
				surrender_progress < 0.40
			}
		}
		GBR = { save_event_target_as = entente_transfer_to }
	}
	else_if = {
		limit = {
			NOT = { tag = CAN }
			CAN = {
				is_in_faction_with = ROOT
				is_subject = no
				surrender_progress < 0.40
			}
		}
		CAN = { save_event_target_as = entente_transfer_to }
	}
	else_if = {
		limit = {
			NOT = { original_tag = NFA }
			NFA = {
				is_in_faction_with = ROOT
				is_subject = no
				surrender_progress < 0.40
			}
		}
		NFA = { save_event_target_as = entente_transfer_to }
	}
	else_if = {
		limit = {
			any_allied_country = {
				is_subject = no
				is_actual_major = yes
				surrender_progress < 0.40
				NOT = {
					tag = AST
					tag = NZL
					tag = GEX
				}
			}
		}
		random_other_country = {
			limit = {
				is_in_faction_with = ROOT
				is_subject = no
				is_actual_major = yes
				surrender_progress < 0.40
				NOT = {
					tag = AST
					tag = NZL
					tag = GEX
				}
			}
			save_event_target_as = entente_transfer_to
		}
	}
	else = {
		save_event_target_as = entente_collapses
	}
}

CAN_entente_collapses = {
	hidden_effect = {
		# release puppets
		every_subject_country = {
			ROOT = {
				set_autonomy = {
					target = PREV
					autonomy_state = autonomy_free
				}
			}
		}
		# inform Entente members
		every_possible_country = { clr_country_flag = KR_was_in_entente }
		every_other_country = {
			limit = { is_in_faction_with = PREV }
			set_country_flag = KR_was_in_entente
			country_event = entente.257
			# Portugal collapses if exiled
			if = {
				limit = { tag = POR }
				country_event = { id = portugal_exile.14 days = 7 }
			}
		}
		# peace out the Entente
		every_country = {
			limit = {
				has_war = yes
				is_valid_peace_target = yes
				NOT = { has_country_flag = KR_was_in_entente }
				any_enemy_country = { has_country_flag = KR_was_in_entente }
			}
			country_event = { id = entente.258 hours = 3 }
		}
		# disband ISAC if created
		if = {
			limit = { is_in_tech_sharing_group = entente_research }
			every_country = {
				limit = { is_in_tech_sharing_group = entente_research }
				remove_from_tech_sharing_group = entente_research
			}
		}
		# disband IEDC if created
		if = {
			limit = { has_global_flag = CAN_iedc_created }
			clr_global_flag = CAN_iedc_created
			event_target:iedc_leader = { remove_ideas = CAN_iedc_main }
			every_country = {
				clr_country_flag = CAN_refused_iedc
				CAN_remove_from_iedc = yes
			}
			clear_global_event_target = iedc_leader
		}
		# remove NFA's claims and cores, if it's dead
		if = {
			limit = { NOT = { country_exists = NFA } }
			every_state = {
				remove_core_of = NFA
				remove_claim_by = NFA
			}
		}
		add_named_threat = {
			threat = 15
			name = CAN_entente_collapse_threat
		}
		IRE = {
			if = {
				limit = { is_in_faction_with = ENT }
				country_event = { id = ireland.2006 days = 15 }		# IRE joining RP?
			}
			else = {
				country_event = { id = irelandflavour.7 days = 7 }		# Fall of IMP - should only be for if IMP is dead, ENT can still be alive
			}
		}
	}
	dismantle_faction = yes
	GBR = { save_global_event_target_as = KR_entente_leader }
}
