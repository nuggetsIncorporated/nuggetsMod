OTT_Suppressing_Society_Category = {
	OTT_suppression_of_society = {
		icon = GFX_decision_freedom
		priority = 1000

		allowed = {
			always = no
		}
		available = {
			has_completed_focus = OTT_Reform_on_the_Homefront
			has_completed_focus = OTT_Lift_Ban_Political_Parties
			if = {
				limit = { NOT = { has_country_flag = ott_free_press } }
				has_completed_focus = OTT_Lift_Press_Restrictions
			}
			NOT = { has_idea = ott_varlik_vargisi }
		}

		is_good = no
		fixed_random_seed = no
		days_mission_timeout = 300

		complete_effect = {
			log = "[GetLogRoot]: Decision complete OTT_suppression_of_society"
			add_political_power = 50
			add_war_support = 0.1
			set_temp_variable = { ideology_change_var = 0.05 }
			change_ruling_party_popularity = yes
			if = {
				limit = { has_country_flag = ott_women_suffrage }
				hidden_effect = { country_event = { id = ottkem.54 days = 50 random_days = 30 } }
			}
		}

		timeout_effect = {
			log = "[GetLogRoot]: Decision timeout OTT_suppression_of_society"
			activate_mission = OTT_suppression_of_society
			if = {
				limit = { has_war = yes }
				random_list = {
					33 = {
						add_stability = -0.01
					}
					33 = {
						add_war_support = -0.01
					}
					34 = {
						add_political_power = -10
					}
					1 = {
						every_owned_state = {
							ott_increase_revolt_risk = yes
						}
					}
				}
			}
			else = {
				random_list = {
					30 = {
						add_stability = -0.10
					}
					30 = {
						add_war_support = -0.10
					}
					30 = {
						add_political_power = -100
					}
					10 = {
						every_owned_state = {
							ott_increase_revolt_risk = yes
						}
					}
				}
			}
		}
	}
