country_event = { #Fate of America
	id = annex.2000
	title = annexation_361
	desc = annexation_361_desc
	picture = GFX_report_event_generic_conference2

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = { NOT = { country_exists = NEE } }
				NEE = { clear_cores_of_tag_if_possible = yes }
			}
			every_state = {
				if = {
					limit = {
						OR = {
							is_mainland_america = yes
							is_american_pacific_islands = yes
							state = 463 #Alaska
						}
					}
					add_core_of = USA
				}
				else = {
					remove_core_of = USA
				}
				remove_core_of = WCA
				remove_core_of = ACC
				remove_core_of = APG
				remove_claim_by = USA
				remove_claim_by = WCA
				remove_claim_by = ACC
				remove_claim_by = APG
			}
			463 = { retain_state = yes } #Alaska
			every_state = {
				limit = { is_american_pacific_islands = yes }
				add_core_of = HAW
				retain_state = yes
			}
			WCA = {
				if = {
					limit = { has_character = WCA_jack_reed }
					retire_character = WCA_jack_reed
				}
			}
			USA = {
				USA_swap_to_blue_cosmetic_tag = yes
				if = {
					limit = { NOT = { has_character = USA_douglas_macarthur } }
					USA_douglas_macarthur = { set_nationality = USA }
					promote_character = { character = USA_douglas_macarthur ideology = paternal_autocrat_subtype }
				}
			}
		}
	}

	option = { #Liberate USA
		log = "[GetLogInfo]: event annex.2000 option annex.2000.a"
		name = annex.2000.a
		trigger = {
			is_subject = no
			has_socialist_government = no
		}
		ai_chance = { base = 60 }
		if = {
			limit = {
				is_ally_with = ENT
				NOT = { country_exists = NEE }
				has_elected_government = yes
			}
			effect_tooltip = {
				set_temp_variable = { tag_to_release = USA }
				release_targeted_tag_nocosmetic = yes
			}
			set_temp_variable = { tag_to_release = NEE }
			hidden_effect = {
				var:tag_to_release = {
					USA_swap_to_colourless_cosmetic_tag = yes
					# Set flags
					set_country_flag = NEE_claimed_america
					set_country_flag = NEE_focusing_Entente
					NEE_united_states_congress = { set_character_name = USA_united_states_congress }
					every_state = {
						limit = {
							is_claim_or_core_of_ROOT = no
							OR = {
								is_mainland_america = yes
								state = 463 #Alaska
							}
						}
						add_core_of = NEE
					}
				}
				release_targeted_tag_nocosmetic = yes
				var:tag_to_release = {
					if = {
						limit = { has_government = social_democrat }
						set_popularities = {
							totalist = 0
							radical_socialist = 4
							syndicalist = 9
							social_democrat = 25
							social_liberal = 18
							market_liberal = 23
							social_conservative = 7
							authoritarian_democrat = 9
							paternal_autocrat = 2
							national_populist = 3
						}
					}
					else_if = {
						limit = { has_government = social_conservative }
						set_popularities = {
							totalist = 0
							radical_socialist = 4
							syndicalist = 9
							social_democrat = 5
							social_liberal = 18
							market_liberal = 23
							social_conservative = 27
							authoritarian_democrat = 9
							paternal_autocrat = 2
							national_populist = 3
						}
					}
					else = {
						set_popularities = {
							totalist = 0
							radical_socialist = 4
							syndicalist = 9
							social_democrat = 5
							social_liberal = 28
							market_liberal = 33
							social_conservative = 7
							authoritarian_democrat = 9
							paternal_autocrat = 2
							national_populist = 3
						}
					}
				}
			}
		}
		else = {
			set_temp_variable = { tag_to_release = USA }
			release_targeted_tag_nocosmetic = yes
			hidden_effect = {
				var:tag_to_release = {
					if = {
						limit = {
							has_government = market_liberal
							has_country_leader = { character = USA_douglas_macarthur }
						}
						random_list = {
							1 = { promote_character = USA_alf_landon }
							1 = { promote_character = USA_wendell_willkie }
						}
					}

						limit = { has_completed_focus = USA_back_in_business }
						uncomplete_national_focus = { focus = USA_back_in_business uncomplete_children = yes }
					}
				}
			}
		}
	}

	option = { #Liberate WCA

		name = annex.2000.b
