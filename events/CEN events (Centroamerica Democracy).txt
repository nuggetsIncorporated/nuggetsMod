			popularity = 0.02
		}
		add_popularity = {
			ideology = social_liberal

		}
		promote_character = {

		}
		hidden_effect = {
			country_event = { id = cendem.5 days = 4 }
		}
	}
	option = { #Choose Jose Maria Moncada
		log = "[GetLogInfo]: event cendem.3 option cendem.3.b"
		name = cendem.3.b
		add_popularity = {
			ideology = market_liberal
			popularity = 0.03
		}

			limit = { HON_jose_maria_moncada_tapia = { has_ideology = market_liberal_subtype } }

				character = HON_jose_maria_moncada_tapia

			}
		}

			add_country_leader_role = {
				character = HON_jose_maria_moncada_tapia

				}
				promote_leader = yes
			}
		}
		hidden_effect = {
			country_event = { id = cendem.5 days = 4 }
		}
	}
}
