country_event = { #Fate of Gabon
	id = annex.5000
	title = annexation_539
	desc = annexation_539_desc
	picture = GFX_report_event_generic_conference2

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			annexations_create_france_event_target = yes
			annexations_create_mittelafrika_event_target = yes
			if = {
				limit = {
					OR = {
						original_tag = MAF
						annexations_france_can_annex_overseas = yes
					}
				}
				save_event_target_as = can_annex
			}
		}
	}

	option = { #Liberate
		log = "[GetLogInfo]: event annex.5000 option annex.5000.a"
		name = annex.5000.a
		trigger = {
			is_subject = no
		}
		ai_chance = {
			base = 40
			modifier = {
				factor = 0
				annexations_can_annex = yes
			}
			modifier = {
				factor = 0
				has_subject = event_target:mittelafrika
			}
			modifier = {
				factor = 0
				annexations_AI_give_to_france_instead = yes
			}
		}
		set_temp_variable = { tag_to_release = GAB }
		release_targeted_tag = yes
	}

	option = { #Give to France
		log = "[GetLogInfo]: event annex.5000 option annex.give_to_france"
		name = annex.give_to_france
		trigger = {
			has_event_target = france
		}
		ai_chance = {
			base = 60
			modifier = {
				factor = 0
				annexations_can_annex = yes
			}
			modifier = {
				factor = 0
				annexations_AI_will_consider_returning_colonies = no
			}
			modifier = {
				factor = 0
				has_subject = event_target:mittelafrika
			}
			modifier = {
				factor = 0
				event_target:france = { annexations_AI_dont_give_to_non_puppet_if_major = yes }
			}
			modifier = {
				factor = 0.1
				event_target:france = { is_subject = yes }
			}
		}
		set_temp_variable = { tag_to_release = event_target:france }
		GAB = { transfer_colonial_territory_to_european_country = yes }
	}

	option = { #Give to MAF
		log = "[GetLogInfo]: event annex.5000 option annex.africa.give_to_mittelafrika"
		name = annex.africa.give_to_mittelafrika
		trigger = {
			has_event_target = mittelafrika
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				has_subject = event_target:mittelafrika
				annexations_can_annex = no
			}
		}
		set_temp_variable = { tag_to_release = event_target:mittelafrika }
		GAB = { set_transfer_target_cores = yes }
		transfer_targeted_territory = yes
	}

		name = annex.give_to_overlord
		trigger = {
			annexations_should_give_lands_to_overlord = yes
		}
