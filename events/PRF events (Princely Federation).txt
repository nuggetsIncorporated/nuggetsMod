				is_ai = yes

			if = {
				limit = { has_country_flag = PRF_ai_princes }
				country_event = { id = prf.18 days = 14 } #Princes
			}
			else_if = {
				limit = { has_country_flag = PRF_ai_osman }
				country_event = { id = prf.19 days = 14 } #Nizam
			}
			else = {
				country_event = { id = prf.20 days = 14 } #Democracy
			}
		}
		else = {
			if = {
				limit = { PRF_princes_winning = yes }
				hidden_effect = {
					country_event = { id = prf.18 days = 14 } #Princes
				}
			}
			else_if = {
				limit = { PRF_osman_winning = yes }
				hidden_effect = {
					country_event = { id = prf.19 days = 14 } #Nizam
				}
			}
			else = {
				hidden_effect = {
					country_event = { id = prf.20 days = 14 } #Democracy
				}
			}
		}
		clr_country_flag = PRF_ai_princes
		clr_country_flag = PRF_ai_osman
