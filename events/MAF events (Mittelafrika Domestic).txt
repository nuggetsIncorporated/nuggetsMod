	id = mittel.2600
	title = mittel.2600.t
	desc = mittel.2600.d

	is_triggered_only = yes

	trigger = {
		is_subject_of = MAF
		event_target:entente_attacker = {
			has_socialist_government = no
			OR = {
				is_faction_leader = yes #ENT
				AND = {
					is_monarchy = yes #CAN/GBR
					is_subject = no
				}
			}
		}
	}

	immediate = {
		if = {
			limit = { GBR = { is_ally_with = event_target:entente_attacker } }
			GBR = { save_event_target_as = entente_attacker }
		}
		else_if = {
			limit = { IMP = { is_ally_with = event_target:entente_attacker } }
			IMP = { save_event_target_as = entente_attacker }
		}
		#if neither, leave to ENT (already set in the on action)
	}

	option = {
		log = "[GetLogInfo]: event mittel.2600 option mittel.2600.a"
		name = mittel.2600.a
		overlord = {
			set_autonomy = {
				target = ROOT
				autonomy_state = autonomy_free
			}
		}
		leave_faction = yes
		peace_out_non_neighbours_after_leaving_faction = yes
		get_current_government_type = yes
		event_target:entente_attacker = {
			set_autonomy = {
				target = ROOT
				autonomy_state = colonial_government
				end_wars = no
			}
		}
		restore_previous_government_type = yes
		every_other_country = {
			limit = {
				is_ally_with = MAF
				OR = {
					tag = MAF
					has_subject = MAF
					is_major = yes
				}
			}
			country_event = mittel.2601
		}
		every_other_country = {
			limit = {
				is_ally_with = event_target:entente_attacker
				OR = {
					is_faction_leader = yes
					is_major = yes
					AND = {
						original_tag = IMP
						is_monarchy = yes #republics don't care about the Empire
					}
					AND = {
						is_commonwealth_tag = yes
						is_monarchy = yes #republics don't care about the Empire
					}
				}
			}
			country_event = mittel.2602
		}
	}
}
